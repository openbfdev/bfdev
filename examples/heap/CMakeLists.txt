add_executable(heap-benchmark benchmark.c)
add_test(heap-benchmark heap-benchmark)

add_executable(heap-selftest selftest.c)
add_test(heap-selftest heap-selftest)

target_link_libraries(
    heap-benchmark
    -Wl,--whole-archive bfdev
    -Wl,--no-whole-archive
)

target_link_libraries(
    heap-selftest
    -Wl,--whole-archive bfdev
    -Wl,--no-whole-archive
)

if(${CMAKE_PROJECT_NAME} STREQUAL "bfdev")
    install(FILES
        benchmark.c
        selftest.c
        DESTINATION
        ${CMAKE_INSTALL_DOCDIR}/examples/heap
    )

    install(TARGETS
        heap-benchmark
        heap-selftest
        DESTINATION
        ${CMAKE_INSTALL_DOCDIR}/bin
    )
endif()
